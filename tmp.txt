square = \value | value * value

'square 5 equals 25'
    test
        (square 5) == 25

(square 5) == 25
    assert

(square 5)
    assert_eq 25

'square of positive numbers'
    eq
        [0, 1, 3, 5]
            map(square)
        [0, 1, 9, 25]
